def simple_hex(num):
    match num:
        case num if (0 <= num | num < 10):
            return str(num)
        case 10:
            return "A"
        case 11:
            return "B"
        case 12:
            return "C"
        case 13:
            return "D"
        case 14:
            return "E"
        case 15:
            return "F"
    return "Invalid input"
def dectohex(initial_num, position = 1, hex=[]):
    index = 0
    num = initial_num
    while (position > 0):
        digi = num // (16**(position-1))
        hex[index] = str(simple_hex(digi))
        num = num - digi * (16**(position-1))
        position -=1
        index +=1
